async function copySectionsToPage(targetPage, sectionNames, sourceSections, rowNumber) {
  console.log('\n=== ROW ' + rowNumber + ' START ===');
  console.log('Target page: ' + targetPage.name);
  console.log('Sections to copy: ' + sectionNames.join(', '));
  
  // Force page switch
  figma.currentPage = targetPage;
  console.log('Current page set to: ' + figma.currentPage.name);
  
  let currentX = CONFIG.PAGE_MARGIN;
  let currentY = CONFIG.PAGE_MARGIN;
  let rowHeight = 0;
  let sectionsInCurrentRow = 0;
  
  for (let i = 0; i < sectionNames.length; i++) {
    const sectionName = sectionNames[i];
    
    console.log('\n--- Section ' + (i + 1) + ': ' + sectionName + ' ---');
    
    // Find section
    const sourceSection = sourceSections.find(function (section) {
      return section.name === sectionName;
    });
    
    if (!sourceSection) {
      console.log('SKIP: Not found in source');
      continue;
    }
    
    console.log('BEFORE CLONE:');
    console.log('  figma.currentPage.name = ' + figma.currentPage.name);
    console.log('  targetPage.name = ' + targetPage.name);
    console.log('  Are they same? ' + (figma.currentPage === targetPage));
    
    try {
      // THE CRITICAL MOMENT - CLONE
      console.log('CLONING NOW...');
      const clonedSection = sourceSection.node.clone();
      
      console.log('AFTER CLONE:');
      console.log('  figma.currentPage.name = ' + figma.currentPage.name);
      console.log('  Clone parent: ' + (clonedSection.parent ? clonedSection.parent.name : 'NO PARENT'));
      console.log('  Are pages still same? ' + (figma.currentPage === targetPage));
      
      // Set position
      clonedSection.x = currentX;
      clonedSection.y = currentY;
      
      console.log('APPENDING TO TARGET PAGE...');
      targetPage.appendChild(clonedSection);
      
      console.log('SUCCESS: ' + sectionName + ' copied');
      
      // Update position
      rowHeight = Math.max(rowHeight, sourceSection.height);
      currentX += sourceSection.width + CONFIG.SECTION_SPACING;
      sectionsInCurrentRow++;
      
    } catch (error) {
      console.log('FAILED: ' + sectionName);
      console.log('Error: ' + error.message);
      console.log('STOPPING ROW');
      throw error;
    }
  }
  
  console.log('\n=== ROW ' + rowNumber + ' COMPLETE ===\n');
}
